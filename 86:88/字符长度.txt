DATA	SEGMENT
STRING	DB	'ABCDEFGHIJ',0DH
COUNT	EQU	$-STRING
LL	DB	?
DATA	ENDS
STACK	SEGMENT	PARA STACK'STACK'
	DB	100 DUP(?)
STACK	ENDS
CODE	SEGMENT
	ASSUME	CS: CODE, DS: DATA, ES: DATA, SS: STACK
START	PROC	FAR
BEGIN:	PUSH	DS
	MOV	AX,0
	PUSH	AX
	MOV	AX,DATA
	MOV	DS,AX
	MOV	ES,AX
	LEA	DI,STRING	;置被搜索串地址指针
	MOV	DL,0		;置串长度初值为 0
	MOV	AL,0DH
	MOV	CX,COUNT十10	;置循环次数大于串长度
AGAIN:	SCASB
	JE	DONE		;找到结束标志，停止
	INC	DL
	DEC	CX		;循环次数减 1
	JNE	AGAIN		;规定的循环次数未完，循环
	JMP	ERROR		;由计数停止循环,出错,转至出错处理程序

DONE:	MOV	LL,DL
	RET	

START	ENDP
CODE	ENDS
	END	BEGIN
