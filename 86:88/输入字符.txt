编写程序，将一串从键盘输入的字符(输入字符数低于20）存入到变量 buffel中，同时将实际输入的宇符数存放在 CL 寄存器中，并在显示器中显示该串字符

STACK1 SEGMENT STACK
	DW 256 DUP(?)
STACK1 ENDS

DATA SEGMENT
MES DB "INPUT a KEY :,AH,ODH, '$';
STRING DB 20DUP(?), ODH,'$';
DATA ENDS

CODE SEGMENT
	ASSUME CS:CODE, DS:DATA,SS:STACK1
START:
    MOV AX, DATA
    MOV DS, AX

    XOR CX, CX               ; 将 CX 寄存器清零
    MOV DX, OFFSET MES       ; 显示 MES 定义的字符
    MOV AH, 09H
    INT 21H

SSS:
    MOV AH, 01H
    INT 21H
    CMP AL, 1BH             ; 检查 AL 是否为 ESC 的 ASCII 值
    JE EXIT                 ; 如果是 ESC，则跳转到 EXIT 标签

    INC CX                  ; 增加 CX 寄存器的值，用于计算输入字符的数量
    MOV BX, OFFSET STRING   ; 将 BX 寄存器设置为 STRING 数组的偏移地址
    MOV [BX], AL            ; 将输入的字符存储在字符串数组中
    INC BX                  ; 增加 BX 寄存器的值，指向下一个数组元素
    JMP SSS                 ; 跳转回 SSS 标签，等待下一个字符输入

EXIT:
    MOV BX, OFFSET STRING   ; 显示 STRING 定义的字符
    MOV AH, 09H
    INT 21H

    MOV AH, 4CH             ; 退出程序
    INT 21H

CODE ENDS
END START
