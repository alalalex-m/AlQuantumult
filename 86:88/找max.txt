NAME SEARCH MAX

DATA	SEGMENT
BUFFER	DW 	X1, X2, ... , Xn
COUNT	EQU 	$ -BUFFER
MAX	DW	?
DATA	ENDS

STACK	SEGMENT PAPA STACK'STACK'
	DB	64 DUP (?)
TOP	EQU	$-STACK
STACK	ENDS

CODE	SEGMENT
START	PROC FAR
	ASSUME CS: CODE, DS: DATA, SS: STACK
BEGIN:	PUSH	DS
	MOV	AX,0
	PUSH	AX
	MOV	AX,DATA
	MOV	DS,AX
	MOV	AX,STACK
	MOV	SS,AX
	MOV	AX,TOP
	MOV	SP,AX
	MOV	CX,COUNT
	LEA	BX,BUFFER
	MOV	AX,[BX]
	INC	BX
	DEC	CX

AGAIN:	CMP	AX,[BX]
	JGE	NEXT
	MOV	AX,[BX]

NEXT:	INC	BX
	LOOP	AGAIN
	
START	ENDP
CODE	ENDS
	END	BEDIN
