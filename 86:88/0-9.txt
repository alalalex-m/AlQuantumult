NAME 0-9

STACK 	SEGMENT PARA STACK'STACK'
	DB 100 DUP (?)
STACK 	ENDS

CODE	SEGMENT
	ASSUME 	CS: CODE, SS: STACK
START 	PROC 	FAR
BEGIN:	PUSH 	DS
	MOV	AX,0
	PUSH	AX
	MOV	BL,0
	PUSH	BX
GOON:	MOV	DL,20H	;把空格宇符->DL
	MOV	AH,2
	INT	21H	;输出空格宇符
	POP	BX
	MOV	AL,BL
	INC	AL
	DAA		;增量后进行十进制调整
	AND	AL,0FH
	MOV	BL,AL
	PUSH 	BX
	OR	AL,30H	;转换为 ASCII 码
	MOV	DL,AL
	MOV	AH,2
	INT	21H	;输出一个0~9的宇符
	MOV	CX,0FFFFH;为便于观察,插人一定的延时

AGAIN: 	DEC	CX
	JNE	AGIAN
	JMP	GOON

START	ENDP	

CODE	ENDS
	END BEGIN
