STACK1  SEGMENT   STACK
        DW        256 DUP(?)
STACK1  ENDS

DATA    SEGMENT   
MES     DB        'INPUT a KEY :',0AH,0DH,'$'
DATA    ENDS

CODE    SEGMENT   
        ASSUME    CS:CODE,DS:DATA
START:  MOV       AX, DATA
        MOV       DS, AX
        MOV       DX, OFFSET MES     ; 显示 MES 定义的字符
        MOV       AH, 09H
        INT       21H

SSS:    MOV       AH, 01H
        INT       21H
        CMP       AL, '8'            ; 检查是否输入的是 '8'
        JNE       NOT_EIGHT
        MOV       AH, 01H
        INT       21H
        CMP       AL, '8'            ; 检查是否输入的是 '8'
        JNE       NOT_EIGHT
        MOV       AH, 01H
        INT       21H
        CMP       AL, '8'            ; 检查是否输入的是 '8'
        JNE       NOT_EIGHT

        ; 连续输入了三个 '8'，显示 "Good number"
        MOV       DX, OFFSET GOOD_NUM
        MOV       AH, 09H
        INT       21H
        JMP       SSS

NOT_EIGHT:
        CMP       AL, 1BH            ; 按下 ESC 键
        JNE       SSS
        JMP       EXIT

EXIT:
        MOV       AH, 4CH            ; 有键返回 DOS
        INT       21H

CODE    ENDS
        END       START

GOOD_NUM  DB        'Good number', 0AH, 0DH, '$'  ; "Good number" 的字符串
